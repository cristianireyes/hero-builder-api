```json
  "paths": {
    "@services": ["src/services"],
    "@services/*": ["src/services/*"],
    "@interfaces": ["src/interfaces"],
    "@interfaces/*": ["src/interfaces/*"],
    "@middlewares": ["src/middlewares"],
    "@middlewares/*": ["src/middlewares/*"],
    "@resolvers": ["src/resolvers"],
    "@resolvers/*": ["src/resolvers/*"],
    "@clients": ["src/clients"],
    "@clients/*": ["src/clients/*"],
    "@plugins": ["src/plugins"],
    "@plugins/*": ["src/plugins/*"],
    "@helpers": ["src/helpers"],
    "@helpers/*": ["src/helpers/*"],
    "@types": ["src/types.ts"],
    "@config": ["src/dotenv-config"],
    "@mocks": ["src/mocks"],
    "@mocks/*": ["src/mocks/*"]
  }
}
```
